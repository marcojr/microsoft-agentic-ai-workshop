Suggested Linux commands after git pull:

1. MCP server dependencies:

2. Orchestrator dependencies:
   cd ../apps/orchestrator-api
   uv venv
   uv pip install -r requirements.txt
   .venv/bin/python -m pytest tests -q

3. Smoke test DraftAgent only:
   cd ../..
   cd apps/orchestrator-api
   PYTHONPATH="../../mcp-server/src:." .venv/bin/python - <<'PY'
   import asyncio
   from src.agents.draft_agent import DraftAgent

   async def main():
       agent = DraftAgent()
       result = await agent.generate_summary(
           customer={"fullName": "John Smith", "email": "john.smith@contoso.com"},
           order={"orderNumber": "WEB-1001", "deliveryStatus": "Delayed", "status": "Shipped"},
           shipment={"shipmentId": "ship-9001", "status": "Delayed"},
           refunds=[{"refundId": "ref-4001", "status": "Pending Approval"}],
           returns=[],
           knowledge=[{"title": "Delivery Delay Policy", "summary": "Approval is required before compensation is communicated."}],
           risk="High",
       )
       print(result["model"])
       print(result["summary"])

   asyncio.run(main())
   PY
